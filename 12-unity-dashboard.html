Overview

Unity is Zeotap's unified Angular SPA that serves as the single product interface for all CDP capabilities. It dynamically loads product modules based on the authenticated user's product entitlements — a customer licensed only for Targeting sees only the Targeting module; a customer licensed for all products sees the full suite.

Unity uses lazy loading extensively: each product module (Targeting, Insights, Studio, Connect, Destinations, DPUI) is a separate Angular module that is only loaded when the user navigates to that product, significantly reducing initial bundle size.

NgRx State Management

Unity uses NgRx (Redux for Angular) for predictable, testable state management across all feature modules. The store is split into feature slices:

// Segment Actions (SegmentsModule)
export const loadSegments = createAction(
  '[Segments] Load Segments',
  props<{ orgId: number; page: number }>()
);

export const loadSegmentsSuccess = createAction(
  '[Segments] Load Segments Success',
  props<{ segments: Segment[]; total: number }>()
);

export const loadSegmentsFailure = createAction(
  '[Segments] Load Segments Failure',
  props<{ error: string }>()
);

export const estimateSegmentSize = createAction(
  '[Segments] Estimate Size',
  props<{ rules: AudienceRule[]; country: string }>()
);
// Segments Reducer
export interface SegmentsState {
  segments: Segment[];
  total: number;
  loading: boolean;
  estimatedSize: number | null;
  error: string | null;
}

export const segmentsReducer = createReducer(
  initialState,
  on(loadSegments, state => ({ ...state, loading: true })),
  on(loadSegmentsSuccess, (state, { segments, total }) => ({
    ...state, loading: false, segments, total
  })),
  on(loadSegmentsFailure, (state, { error }) => ({
    ...state, loading: false, error
  })),
  on(estimateSegmentSizeSuccess, (state, { size }) => ({
    ...state, estimatedSize: size
  }))
);
// Segments Effects (API calls)
@Injectable()
export class SegmentsEffects {
  loadSegments$ = createEffect(() =>
    this.actions$.pipe(
      ofType(loadSegments),
      switchMap(({ orgId, page }) =>
        this.segmentsApi.list(orgId, page).pipe(
          map(resp => loadSegmentsSuccess({
            segments: resp.segments,
            total: resp.total
          })),
          catchError(err => of(loadSegmentsFailure({ error: err.message })))
        )
      )
    )
  );

  estimateSize$ = createEffect(() =>
    this.actions$.pipe(
      ofType(estimateSegmentSize),
      debounceTime(300),              // debounce UI typing
      switchMap(({ rules, country }) =>
        this.estimatorApi.estimate(rules, country).pipe(
          map(r => estimateSegmentSizeSuccess({ size: r.estimated_size }))
        )
      )
    )
  );
}

Routing & Authentication Guards

// app.routing.ts — lazy-loaded routes
export const appRoutes: Routes = [
  {
    path: PRODUCTS.ZAT.path,           // 'targeting'
    loadChildren: './containers/+segments#SegmentsModule',
    canActivateChild: [AuthGuard],
    data: { product: PRODUCTS.ZAT.name, permission: '' }
  },
  {
    path: PRODUCTS.INSIGHTS.path,      // 'insights'
    loadChildren: './containers/+insights#InsightsModule',
    canActivateChild: [AuthGuard],
    data: { product: PRODUCTS.INSIGHTS.name }
  },
  {
    path: PRODUCTS.STUDIO.path,        // 'studio'
    loadChildren: './containers/+studio#StudioModule',
    canActivateChild: [AuthGuard]
  },
  {
    path: PRODUCTS.DESTINATIONS.path,  // 'destinations'
    loadChildren: './containers/+channel-platform#ChannelPlatformModule',
    canActivateChild: [AuthGuard]
  },
  { path: 'error', component: ErrorComponent },
  { path: '', redirectTo: getInitialRoutingProduct(), pathMatch: 'full' }
];

// AuthGuard — checks RBAC permissions
@Injectable({ providedIn: 'root' })
export class AuthGuard implements CanActivateChild {
  canActivateChild(route: ActivatedRouteSnapshot): boolean {
    const prodInfo = JSON.parse(localStorage.getItem('prod_info') || '{}');
    const product  = route.data['product'];
    return !!prodInfo[product];   // returns true if product is licensed
  }
}

ZUI Design System

The ZUILibrary is Zeotap's internal Angular component library — a design system that enforces visual consistency across Unity and all future Zeotap products.

🎨

Design Tokens

CSS custom properties for brand colors (--zui-primary, --zui-accent), typography scale, spacing grid, border-radius system.

🧩

Core Components

ZuiButton, ZuiInput, ZuiSelect, ZuiModal, ZuiTable, ZuiPaginator, ZuiToast — all with accessible ARIA attributes.

📊

Chart Components

ZuiBarChart, ZuiLineChart, ZuiDonutChart — wrapping ECharts/D3.js with Zeotap brand theming.

🔧

Form Controls

ReactiveFormsModule-compatible controls for rule builders, date pickers, multi-selects with search, and tag inputs.
